## Code Example Block

<CodeExampleBlock
  title="API Integration Example"
  description="Example code for integrating with the API"
  tabs={[
    {
      title: "JavaScript",
      language: "javascript",
      code: `
// Initialize the API client
const client = new ApiClient({
  apiKey: process.env.API_KEY,
  baseUrl: 'https://api.example.com/v1'
});

// Fetch token data
async function getTokenData() {
  try {
    const response = await client.tokens.getAll();
    return response.data;
  } catch (error) {
    console.error('Error fetching token data:', error);
    throw error;
  }
}
      `
    },
    {
      title: "Python",
      language: "python",
      code: `
# Initialize the API client
client = ApiClient(
    api_key=os.environ.get("API_KEY"),
    base_url="https://api.example.com/v1"
)

# Fetch token data
def get_token_data():
    try:
        response = client.tokens.get_all()
        return response.data
    except Exception as error:
        print(f"Error fetching token data: {error}")
        raise
      `
    }
  ]}
